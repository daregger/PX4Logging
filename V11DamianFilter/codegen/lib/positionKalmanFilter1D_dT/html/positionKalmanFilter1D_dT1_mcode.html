1function  [x_aposteriori,P_aposteriori]=positionKalmanFilter1D_dT(dT,x_aposteriori_k,P_aposteriori_k,u,z,gps_update,Q,R,thresh,decay)
 2    %dynamics
 3    A = [1 dT -0.5*dT*dT;
 4         0 1 -dT;
 5         0 0 1];
 6    B = [0.5*dT*dT; dT; 0];
 7    C = [1 0 0];
 8    %prediction
 9    x_apriori=A*x_aposteriori_k+B*u;
10    P_apriori=A*P_aposteriori_k*A'+Q;
11    if abs(u)<thresh
12        x_apriori(2)=decay*x_apriori(2);
13    end
14    %update
15    if gps_update==1
16        y=z-C*x_apriori;
17        S=C*P_apriori*C'+R;
18        K=(P_apriori*C')/S;
19        x_aposteriori=x_apriori+K*y;
20        P_aposteriori=(eye(size(P_apriori))-K*C)*P_apriori;
21    else
22        x_aposteriori=x_apriori;
23        P_aposteriori=P_apriori;
24    end
25end
