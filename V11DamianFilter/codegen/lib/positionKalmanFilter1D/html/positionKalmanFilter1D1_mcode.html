1function  [x_aposteriori,P_aposteriori]=positionKalmanFilter1D(A,B,C,x_aposteriori_k,P_aposteriori_k,u,z,gps_update,Q,R,thresh,decay)
 2    %prediction
 3    x_apriori=A*x_aposteriori_k+B*u;
 4    P_apriori=A*P_aposteriori_k*A'+Q;
 5    if abs(u)<thresh
 6        x_apriori(2)=decay*x_apriori(2);
 7    end
 8    %update
 9    if gps_update==1
10        y=z-C*x_apriori;
11        S=C*P_apriori*C'+R;
12        K=(P_apriori*C')/S;
13        x_aposteriori=x_apriori+K*y;
14        P_aposteriori=(eye(size(P_apriori))-K*C)*P_apriori;
15    else
16        x_aposteriori=x_apriori;
17        P_aposteriori=P_apriori;
18    end
19end
